00030  * Description:
00031  * Common macro definitions for bit field support. The macros are introduced to get rid 
00032  * of read-modify-write accesses and to avoid tool dependent usage of C constructs (bit 
00033  * field structs).
00034  * The macros require another 3 predefined macros per bitfield which describe the 
00035  * bitfield layout: 
00036  * <PERIPHERAL>_<BITFIELD>_OFFSET_ADDRESS
00037  * <PERIPHERAL>_<BITFIELD>_BIT_OFFSET
00038  * <PERIPHERAL>_<BITFIELD>_BITFIELD_SIZE
00039  * These predefined macros together with get- or set-macros then allow straight forward, 
00040  * object oriented, peripheral specific macro definitions with encapsulated access type.
00041  * Example: 
00042  * #define RER1_BF1_OFFSET_ADDRESS       4
00043  * #define RER1_BF1_OFFSET_BIT_OFFSET    1
00044  * #define RER1_BF1_OFFSET_BITFIELD_SIZE 1
00045  * #define RER1_BF2_OFFSET_ADDRESS       4
00046  * #define RER1_BF2_OFFSET_BIT_OFFSET    2
00047  * #define RER1_BF2_OFFSET_BITFIELD_SIZE 4
00048  * #define RER1_BF1_SET(moduleBaseAddress, value) \
00049  *             SET8_BITFIELD_N(moduleBaseAddress, PER1_BF1, value)
00050  * Note: Bitfields which are alone within their container (byte, halfword or word) should 
00051  * use a BITFIELD_SIZE equal to the container size to allow automatic compiler 
00052  * optimisation.
00053  *
00054  *****************************************************************************************
00055  */

00072 /* Bit offset relative to the byte aligned bit field address. */
00073 #define BIT_OFFSET(bitfieldName)  (bitfieldName ## _BITOFFSET % 8)
00074 /* Byte offset relative to the given bit field offset addres. 
00075  * 8 * BYTE_OFFSET(bitfieldName) + BIT_OFFSET(bitfieldName) = BIT_OFFSET_ ## bitfieldName */
00076 #define BYTE_OFFSET(bitfieldName) (bitfieldName ## _BITOFFSET / 8)
00077 
00078 /* Byte aligned bit field offset address. */
00079 #define BYTE_OFFSET_ADDRESS(bitfieldName) (bitfieldName ## _OFFSETADDRESS + BYTE_OFFSET(bitfieldName))
00080 /* Halfword aligned bit field offset address. */
00081 #define HALFWORD_OFFSET_ADDRESS(bitfieldName) (BYTE_OFFSET_ADDRESS(bitfieldName) & ~0x01)
00082 /* Word aligned bit field offset address. */
00083 #define WORD_OFFSET_ADDRESS(bitfieldName) (BYTE_OFFSET_ADDRESS(bitfieldName) & ~0x03)
00084 
00085 /* Bit offset for the byte aligned bit field offset address. */
00086 #define BYTE_BIT_OFFSET(bitfieldName) \
00087             (BIT_OFFSET(bitfieldName) + (8 * (BYTE_OFFSET_ADDRESS(bitfieldName) & 0x00)))
00088 /* Bit offset for the halfword aligned bit field offset address. */
00089 #define HALFWORD_BIT_OFFSET(bitfieldName) \
00090             (BIT_OFFSET(bitfieldName) + (8 * (BYTE_OFFSET_ADDRESS(bitfieldName) & 0x01)))
00091 /* Bit offset for the word aligned bit field offset address. */
00092 #define WORD_BIT_OFFSET(bitfieldName) \
00093             (BIT_OFFSET(bitfieldName) + (8 * (BYTE_OFFSET_ADDRESS(bitfieldName) & 0x03)))
00094 
00095 /* Value conversion from a plain value to the correctly bit aligned byte value. (DeNormalisation) */
00096 #define BYTE_VALUE_DN(bitfieldName, value)     (((UINT8)  value) << BYTE_BIT_OFFSET(bitfieldName))
00097 /* Value conversion from a plain value to the correctly bit aligned halfword value. (DeNormalisation) */
00098 #define HALFWORD_VALUE_DN(bitfieldName, value) (((UINT16) value) << HALFWORD_BIT_OFFSET(bitfieldName))
00099 /* Value conversion from a plain value to the correctly bit aligned word value. (DeNormalisation) */
00100 #define WORD_VALUE_DN(bitfieldName, value)     (((UINT32)  value) << WORD_BIT_OFFSET(bitfieldName))

00102 /* Normalized bit field mask. */
00103 #define MASK(bitfieldName)          (bitfieldName ## _BITFIELDSIZE == 32 ? ~0 : (((UINT32) 1) << bitfieldName ## _BITFIELDSIZE) - 1)
00104 /* Bitfield mask used with byte accesses. (denormalized) */
00105 #define BYTE_MASK(bitfieldName)     BYTE_VALUE_DN(bitfieldName, MASK(bitfieldName))
00106 /* Bitfield mask used with halfword accesses. (denormalized) */
00107 #define HALFWORD_MASK(bitfieldName) HALFWORD_VALUE_DN(bitfieldName, MASK(bitfieldName))
00108 /* Bitfield mask used with word accesses. (denormalized) */
00109 #define WORD_MASK(bitfieldName)     WORD_VALUE_DN(bitfieldName, MASK(bitfieldName))
