00426 PUBLIC INLINE UINT32  
00427 GSN_IPv6_ADDR_TYPE(UINT32 *a)
00428 {
00429         UINT32 temp;
00430 
00431         if((a[0] & 0xFF000000)==0xFF000000)
00432         {
00433                 return GSN_IPv6_MULTICAST;
00434         }
00435 
00436         if((a[0] & 0xFE800000)==0xFE800000) 
00437         {
00438                 return GSN_IPv6_LINK_LOCAL;
00439         }
00440         
00441         temp = a[0] | a[1] | a[2];
00442         if(temp == 0)
00443         {
00444                 if(a[3] == 1)
00445                 {
00446                         return GSN_IPv6_LOOPBACK;
00447                 }
00448                 if(a[3] == 0)
00449                 {
00450                         return GSN_IPv6_UNSPECIFIED;
00451                 }
00452         }
00453
